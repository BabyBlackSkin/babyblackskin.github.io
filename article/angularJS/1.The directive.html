AngularJS 1.x - The directive ​

在使用AngularJS的时候，指令是无处不在的，指令可以用来拓展element，attribute。丰富DOM元素的功能，常用的例如：ng-click,ng-repeat。

  • ng-click指令可以让一个元素监听click事件，
  • ng-repeat指令用来循环遍历数组，来生成元素。 ng-click和原生的click有什么区别，为什么不用原生的click呢？下面举一个例子

点击按钮修改p标签内的文字

Html

html
<html>
    <body ng-controller="IndexCtrl as ctrl">
        <p id="title">{{'标题：' + ctrl.title}}</p>
        <button ng-click="ctrl.clickOne()">Button A</button>
        <button id="btnTwo">Button B</button>
    </body>
</html>

JS

javascript

<script>
// 原生js
    document.getElementById('btnTwo').addEventListener('click', function () {
        document.getElementById('title').innerHTML='标题：clickTwo'
    })
// angularjs
    angular.controller('IndexCtrl', ['$route', function IndexCtrl($route) {
        this.title = ""
        this.clickOne = function () {
            this.title = "clickOne"
        }
    }])
</script>

可以看出，当使用angularJs时，它会自动的进行脏监测，当数据发送变化时，通知视图重新渲染。实际上是将执行的内容放到$apply中进行，最终通知$digest进行循环，通知所有的watcher

创建一个简单的指令 ​

html

<html>
    <body>
        <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
            <custom-directive></custom-directive>
        </section>
    </body>
</html>

<script>
    // angularjs
    angular.controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {
            }])
            .directive('customDirective',function(){
                return{
                    restrict:'E',
                    replace:true,
                    template:'<div>我是一个自定义的组件</div>'
                }
            })

</script>

F12查看Dom ​

这是一个非常简单的指令，可以在页面上进行复用，假设template是一个特别复杂的dom结构，通过指令可以非常轻松的复用，避免重复编写

指令属性 ​

上面只是一个简单的指令，指令中还有其他很多的属性，用于实现复杂的功能。下面是一个directive的完整模板结构

javascript
angular.module('app').directive('directiveName', function () {
    return {
        restrict: String,
        priority: Number,
        terminal: Boolean,
        template: ' String or Template Function',
        templateUrl: String,
        replace: 'Boolean or String',
        scope: 'Boolean or Object',
        transclude: Boolean,
        controller: function (scope, element, attrs, transclude, otherInjectables) {},
        controllerAs: String,
        require: String,
        link: function (scope, iElement, iAttrs) {},
        compile: function (tElement, tAttrs, transclude) {
            return {
                pre: function (scope, iElement, iAttrs, controller) {},
                post: function (scope, iElement, iAttrs, controller) {}
            };
            //或 return function postLink() {} 
        }
    };
}

restrict限制；约束；

表示指令在dom中以什么样式形式进行声明，是一个可选值，范围：E表示元素，A表示属性，C表示class，M表示注释。默认为EA

html
<!--E-->
<custom-directive></custom-directive>
<!--a-->
<div custom-directive></div>
<!--C-->
<div class="custom-directive"></div>
<!--M-->
<!--custom-directive-->

priority优先级

指令的优先级，默认为0，数字越大优先级越高，数字越小优先级越低优先级越高的指令越先被加载（仅在DOM属性时起作用）

javascript
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <div ten-priority twenty-priority></div>     
    </section>
</html>

<script>
    angular.module('app').directive('tenPriority', function () {
        return {
            restrict: 'EACM',
            priority: 10,
            controller: function () {
                console.log('我的优先级是10')
            }
        }
    })
    .directive('twentyPriority', function () {
        return {
            restrict: 'EACM',
            priority: 20,
            controller: function () {
                console.log('我的优先级是20')
            }
        }
    })

</script>

terminal

terminal为boolean值为boolean值，用于决定优先级比自己低的指令是否还需要执行，以上方例子为里，为twentyPriority的terminal:true

template模板

template为字符串或者函数，用于定义指令的模板，如果为字符串，则直接使用字符串作为模板，如果为函数，则使用函数返回的字符串作为模板。在第一个例子中采用了字符串作为template的值，下面展示一个函数作为template的值的指令 template函数接受两个参数。第一个参数是使用此指令的dom，第二个参数是使用此指令dom上所有的属性

html
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <custom-directive name="AngularJs"></custom-directive>
    </section>
</html>
<script>
    angular.controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {}])
            .directive('customDirective', function () {
                return {
                    restrict: 'E',
                    replace: true,
                    template: function (element, attrs){
                        console.log(element)
                        console.log(attrs)
                        return `<p>Hello World, I am ${attrs.name}</p>`
                    }
                }
    })

</script>

templateUrl模板路径

templateUrl可以将模板以文件的形似单独进行维护。同样也是支持函数的形式。

html
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <custom-directive name="AngularJs"></custom-directive>
    </section>
</html>
<script>
    angular.controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {}])
            .directive('customDirective', function () {
                return {
                    restrict: 'E',
                    templateUrl: 'template/echo-template.html'
                }
    })

</script>

replace替换

replace值为布尔值，用于决定指令模板是否替换声明指令的DOM元素，默认为false

relace为true

replace为false

scope作用域 scope属性用于决定指令作用域与父级作用域的关系，可选值有布尔值或者一个对象，默认为false 当 scope:false 时，表示指令不创建额外的作用域，默认继承使用父级作用域，指令中能正常使用和修改父级中所有变量和方法

javascript
<html>
  <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
    我是父<input ng-model="inputValue">
    <custom-directive></custom-directive>
  </section>
</html>
<script>
  app .controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {
    $scope.inputValue="HelloWord"
  }])
    .directive('customDirective', function () {
      return {
        restrict: 'E',
        scope:false,
        // replace: true,
        template: '我是子<input ng-model="inputValue">'
      }
})
</script>

当 scope:true 时表示指令创建自己的作用域，但初始时，仍然会继承父作用域，只是后续的修改是不会影响父作用域的

javascript
<html>
  <div>
    <p>我是父</p>
    姓名<input ng-model="name">
    年龄<input ng-model="age">
  </div>
  <custom-directive></custom-directive>
</html>
<script>
  app.controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {
  $scope.name="zhangsan"

}])
  .directive('customDirective', function () {
    return {
        restrict: 'E',
        scope:true,
        template: `
                <div>
                    <p>我是子</p>
                    姓名<input ng-model="name">
                    年龄<input ng-model="age">
                </div>`,
        controller:function ($scope){
            $scope.age = 18
        }
    }
})
</script>

可以看到 在子作用域中 name是继承自父作用域的，且修改父作用域时，子作用域同步修改。但一旦修改了子作用域，则父作用域无法再影响子作用域了； age是子作用域独立的属性，没有继承自父作用域，所以父子之间没有关联

在某些情况被隔离的作用域就是需要父作用域的数据，如何实现？此时就可以使用scope:{}的方式，来指定，哪些属性需要双向绑定，哪些是单向的。在AngularJS中，支持三种绑定策略：@、=、&

@单向传递，且类型必须是字符串，父能影响子，子不影响父。

javascript
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <div>
            <p>我是父</p>
            姓名<input ng-model="name">
            年龄<input ng-model="age">
        </div>
        <custom-directive name="{{name}}"></custom-directive>
    </section>

</html>
<script>
    app.controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {
    $scope.name="zhangsan"

}])
    .directive('customDirective', function () {
    return {
    restrict: 'E',
    scope: {
        name:"@",
    },
    template: `
                <div>
                    <p>我是子</p>
                    姓名<input ng-model="name">
                    年龄<input ng-model="age">
                </div>`,
    controller:function ($scope){
    $scope.age = 18
}
}
})
</script>

可以看到 在子作用域中 name是继承自父作用域的，且修改父作用域时，子作用域同步修改。且子作用域后，在修改父作用域仍然会更新子作用域

=双向数据绑定。用于传递各类数据，字符串，对象，数组等等。无论是修改父还是修改子，都会同步影响

javascript
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <div>
            <p>我是父</p>
            姓名<input ng-model="name">
            年龄<input ng-model="age">
        </div>
        <custom-directive name="name"></custom-directive>
    </section>
</html>
<script>
    app.controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {
    $scope.name="zhangsan"

}])
    .directive('customDirective', function () {
        return {
        restrict: 'E',
        scope: {
            name:"=",
        },
        template: `
                    <div>
                        <p>我是子</p>
                        姓名<input ng-model="name">
                        年龄<input ng-model="age">
                    </div>`,
        controller:function ($scope){
            $scope.age = 18
        }
    }
})
</script>

&用于传递父作用域中声明的方法，也就是通过&我们可以在指令中直接使用父的方法

javascript
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <custom-button click-handle="sayHello()"></custom-button>
    </section>
</html>
<script>
    app
    .controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {

        $scope.sayHello = function (){
            console.log("我是父级作用域")
        }

     }])
    .directive('customButton', function () {
        return {
            restrict: 'E',
            scope: {
                clickHandle:"&",
            },
            template: `<button ng-click="clickHandle()">点击</button>`,
            controller: function () {
            }
        }
    })
})
</script>

transclude

支持传入一个自定义的dom，类型Vue中的插槽

javascript
<html>
    <section class="mob-doc content" ng-controller="BlogCtrl as ctrl">
        <custom-directive>
            <span>我是body</span>
        </custom-directive>
    </section>
</html>
<script>
    app
    .controller('BlogCtrl', ['$scope', function CheckBoxCtrl($scope) {

     }])
    .directive('customDirective', function () {
        return {
            restrict: 'E',
            transclude:true,
            template: `
                <div>
                    <div class="container" ng-transclude></div>
                </div>`,
        }
    })
})
</script>
