NettyServer.java ​

package com.babyblackskin.netty;

import io.netty.bootstrap.ServerBootstrap;
import io.netty.channel.ChannelFuture;
import io.netty.channel.ChannelOption;
import io.netty.channel.EventLoopGroup;
import io.netty.channel.nio.NioEventLoopGroup;
import io.netty.channel.socket.nio.NioServerSocketChannel;

/**
 * @author qin.yu.bin
 * @version V1.0.0
 * @ClassName NettyServer
 * @Description
 * @date 2024/7/16 14:29
 */
public class NettyServer {
    public static void main(String[] args) {
        new NettyServer().bind(7397);
    }


    private void bind(int port) {
        // 配置服务端的NIO线程组
        EventLoopGroup parentGroup = new NioEventLoopGroup();
        EventLoopGroup childGroup = new NioEventLoopGroup();

        try {
            ServerBootstrap b = new ServerBootstrap();
            b.group(parentGroup, childGroup)
                    // 非阻塞
                    .channel(NioServerSocketChannel.class)
                    // channel策略为SO_BACKLOG，当处理Accept的速率小于连接建立的速率时，全连接队列中堆积的连接数大于SO_BACKLOG设置的值是，便会抛出异常
                    .option(ChannelOption.SO_BACKLOG, 128)
                    // 自定义处理Handler
                    .childHandler(new CustomChannelInitializer());
            ChannelFuture f = b.bind(port).sync();
            f.channel().closeFuture().sync();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            childGroup.shutdownGracefully();
            parentGroup.shutdownGracefully();
        }
    }
}

CustomChannelInitializer.java ​

LineBasedFrameDecoder解码器，默认以换行符作为分隔符。（TCP中的半包粘包问题）

java
package com.babyblackskin.netty;

import io.netty.channel.ChannelInitializer;
import io.netty.channel.socket.SocketChannel;
import io.netty.handler.codec.DelimiterBasedFrameDecoder;
import io.netty.handler.codec.Delimiters;
import io.netty.handler.codec.FixedLengthFrameDecoder;
import io.netty.handler.codec.LineBasedFrameDecoder;
import io.netty.handler.codec.string.StringDecoder;
import io.netty.handler.codec.string.StringEncoder;

import java.nio.charset.Charset;
import java.nio.charset.StandardCharsets;

/**
 * @author qin.yu.bin
 * @version V1.0.0
 * @ClassName MyChannelInitializer
 * @Description
 * @date 2024/7/16 14:41
 */
public class CustomChannelInitializer extends ChannelInitializer<SocketChannel> {
    @Override
    protected void initChannel(SocketChannel socketChannel) throws Exception {
        // 解码器
        // 基于换行符必须以换行符结尾，才能解码， 否则会丢弃，不会调用channelRead
        socketChannel.pipeline().addLast(new LineBasedFrameDecoder(1024));
        // 基于指定字符串，指定为换行符，这就等同于  LineBasedFrameDecoder
        // socketChannel.pipeline().addLast(new DelimiterBasedFrameDecoder(1024, false, Delimiters.lineDelimiter()));
        // 基于最大长度
        // socketChannel.pipeline().addLast(new FixedLengthFrameDecoder(4));

        // 解码转String
        socketChannel.pipeline().addLast(new StringDecoder(Charset.forName("GBK")));
        // String编码
        socketChannel.pipeline().addLast(new StringEncoder(StandardCharsets.UTF_8));


        // 在管道中添加我们自己的接收数据实现方法
        socketChannel.pipeline().addLast(new CustomServerHandler());
    }
}

CustomServerHandler.java ​

java
package com.babyblackskin.netty;

import io.netty.buffer.ByteBuf;
import io.netty.buffer.Unpooled;
import io.netty.channel.Channel;
import io.netty.channel.ChannelHandlerContext;
import io.netty.channel.ChannelInboundHandlerAdapter;
import io.netty.channel.socket.nio.NioSocketChannel;

import java.net.InetSocketAddress;
import java.nio.channels.SocketChannel;
import java.nio.charset.Charset;
import java.nio.charset.StandardCharsets;
import java.text.SimpleDateFormat;
import java.util.Date;

/**
 * @author qin.yu.bin
 * @version V1.0.0
 * @ClassName MyServerHandler
 * @Description
 * @date 2024/7/16 15:07
 */
public class CustomServerHandler extends ChannelInboundHandlerAdapter {
    @Override
    public void channelActive(ChannelHandlerContext ctx) throws Exception {

        NioSocketChannel sc = (NioSocketChannel) ctx.channel();
        System.out.println("链接报告开始");
        System.out.println("链接报告信息：有一个客户端连接到本服务端");
        System.out.println("链接报告IP：" + sc.localAddress().getHostString());
        System.out.println("链接报告PORT" + sc.localAddress().getPort());
        System.out.println("链接报告完毕");

    }

    /**
     * 当客户端主动断开服务端的链接后，这个通道就是不活跃的。也就是说客户端与服务端的关闭了通信通道并且不可以传输数据
     */
    @Override
    public void channelInactive(ChannelHandlerContext ctx) throws Exception {
        System.out.println("channelInactive");
    }

    /**
     * 底层的socket读到了bytes就会被触发一次
     */
    @Override
    public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
        System.out.println("readComplete");
    }

    /**
     * 抓住异常，当发生异常的时候，可以做一些相应的处理，比如打印日志、关闭链接
     */
    @Override
    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
        ctx.close();
        System.out.println("异常信息：\r\n" + cause.getMessage());
    }

    /**
     * 会在消息被解码后才会被调用；
     */
    @Override
    public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
        // ChannelInitialize中已经指定了解码器的编码此处无需在进行解码
        // ByteBuf buf = (ByteBuf) msg;
        // byte[] msgByte = new byte[buf.readableBytes()];
        // buf.readBytes(msgByte);
        //
        // System.out.println(new Date() + "接收到信息： " + new String(msgByte, Charset.forName("GBK")));


        // 接收msg消息，此处已经不需要自己进行解码
        System.out.println(new SimpleDateFormat("yyyy-MM-dd HH:mm:ss").format(new Date()) + " 接收到消息：" + msg);

        // ChannelInitialize中已经指定了编码器，此处无需在进行编码
        // String str = "服务器收到：" + new Date() + " " + msg + "\r\n";
        // ByteBuf buf = Unpooled.buffer(str.getBytes().length);
        // buf.writeBytes(str.getBytes(StandardCharsets.UTF_8));
        // ctx.writeAndFlush(buf);


        String str = "服务器收到：" + new Date() + " " + msg + "\r\n";

        // 点对点发送
        ctx.writeAndFlush(str);
    }
}

package com.babyblackskin.netty;

import io.netty.buffer.ByteBuf;
import io.netty.buffer.Unpooled;
import io.netty.channel.Channel;
import io.netty.channel.ChannelHandlerContext;
import io.netty.channel.ChannelInboundHandlerAdapter;
import io.netty.channel.socket.nio.NioSocketChannel;

import java.net.InetSocketAddress;
import java.nio.channels.SocketChannel;
import java.nio.charset.Charset;
import java.nio.charset.StandardCharsets;
import java.text.SimpleDateFormat;
import java.util.Date;

/**
 * @author qin.yu.bin
 * @version V1.0.0
 * @ClassName MyServerHandler
 * @Description
 * @date 2024/7/16 15:07
 */
public class CustomServerHandler extends ChannelInboundHandlerAdapter {
    @Override
    public void channelActive(ChannelHandlerContext ctx) throws Exception {
        //当有客户端链接后，添加到channelGroup通信组
        ChannelHandler.channelGroup.add(ctx.channel());

        NioSocketChannel sc = (NioSocketChannel) ctx.channel();
        System.out.println("链接报告开始");
        System.out.println("链接报告信息：有一个客户端连接到本服务端");
        System.out.println("链接报告IP：" + sc.localAddress().getHostString());
        System.out.println("链接报告PORT" + sc.localAddress().getPort());
        System.out.println("链接报告完毕");

    }

    /**
     * 当客户端主动断开服务端的链接后，这个通道就是不活跃的。也就是说客户端与服务端的关闭了通信通道并且不可以传输数据
     */
    @Override
    public void channelInactive(ChannelHandlerContext ctx) throws Exception {
        System.out.println("channelInactive");
        // 当有客户端退出后，从channelGroup中移除。
        ChannelHandler.channelGroup.remove(ctx.channel());
    }

    /**
     * 底层的socket读到了bytes就会被触发一次
     */
    @Override
    public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {

        ctx.close();
        System.out.println("异常信息：\r\n" + cause.getMessage());
    }

    /**
     * 会在消息被解码后才会被调用；
     */
    @Override
    public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
        // ChannelInitialize中已经指定了解码器的编码此处无需在进行解码
        // ByteBuf buf = (ByteBuf) msg;
        // byte[] msgByte = new byte[buf.readableBytes()];
        // buf.readBytes(msgByte);
        //
        // System.out.println(new Date() + "接收到信息： " + new String(msgByte, Charset.forName("GBK")));


        // 接收msg消息，此处已经不需要自己进行解码

        // ByteBuf buf = Unpooled.buffer(str.getBytes().length);
